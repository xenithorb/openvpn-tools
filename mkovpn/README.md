#### Some notes about $home_config

Required changes for OpenVPN on EdgeRouter:
```diff
[edit firewall]
+name OVPN_IN {
+    default-action drop
+    rule 1 {
+        action accept
+        destination {
+            address 192.168.1.0/24
+        }
+    }
+}
+name OVPN_LOCAL {
+    default-action drop
+    rule 1 {
+        action accept
+        destination {
+            port 53
+        }
+        protocol udp
+    }
+}
[edit firewall name WAN_LOCAL]
+rule 33 {
+    action accept
+    destination {
+        port 1194
+    }
+    protocol udp
+    state {
+        new enable
+    }
+}
[edit interfaces]
+openvpn vtun0 {
+    config-file /config/openvpn/vpn.example.com.conf
+    firewall {
+        in {
+            name OVPN_IN
+        }
+        local {
+            name OVPN_LOCAL
+        }
+    }
+}
[edit service nat]
+rule 1000 {
+    description "443 Redirect for OpenVPN"
+    destination {
+        port 443
+    }
+    inbound-interface eth0
+    inside-address {
+        port 1194
+    }
+    protocol udp
+    type destination
+}
```
